93 currentSet[nextFlagBit] = rf;

95 obj.size += prop.size;
96 nextFlagBit <<= 1;
97 });

99 if (!currentSet.empty())
100 obj.fields.push_back(std::move(currentSet));

104 // ------------------------------------------
